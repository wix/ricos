Code Block Plugin

Code Block plugin allows the user to show colored code snippets

<Add a code block>

/* Editor */
import { RicosEditor } from 'ricos-editor';
import { pluginCodeBlock } from 'wix-rich-content-plugin-code-block';
import 'ricos-editor/dist/styles.min.css';
import 'wix-rich-content-plugin-code-block/dist/styles.min.css';
<RicosEditor plugins={[pluginCodeBlock()]} />;
/* Viewer */
import { RicosViewer } from 'ricos-viewer';
import { pluginCodeBlock } from 'wix-rich-content-plugin-code-block/dist/module.viewer';
import 'wix-rich-content-plugin-code-block/dist/styles.min.css';
<RicosViewer plugins={[pluginCodeBlock()]} />;
